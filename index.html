packages = [ "qrcode", "matplotlib", ] import qrcode from matplotlib import pyplot as plt from matplotlib import image as mpimg qr = qrcode.QRCode( version=1, error_correction=qrcode.constants.ERROR_CORRECT_L, box_size=10, border=2, ) qr.add_data('0101046440') img = qr.make_image(fill_color="#000000", back_color="white") plt.imshow(img) plt.axis('off') plt.show() display(img, target="mpl")
